program simple_control;

var
    x, y, i : int;

main {
    x = 10;
    y = 5;
    i = 1;
    
    print("Comparando x y y:");
    if x > y {
        print("x es mayor");
    } else {
        print("y es mayor o igual");
    };
    
    print("Contando:");
    while i < 4 do {
        print(i);
        i = i + 1;
    };
    
    print("Fin");
}
end
